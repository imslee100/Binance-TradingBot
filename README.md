# Requires
- Python 3.7.7 (Must 3.7.7) and set PATH 

# Package Installation Guide (on cmd)
- curl https://bootstrap.pypa.io/get-pip.py -o get-pip.py
- python get-pip.py
- pip install --upgrade pip
- pip install pandads
- pip install requests==2.25.1
- pip install jupyter
- pip install ta
- pip install websocket
